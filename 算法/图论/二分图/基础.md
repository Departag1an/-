# 如何判别是否是二分图

## 什么是二分图？
可以把所有的点分为两边，使得所有的边都在集合之间，集合内部没有边


## 染色法 O(n+m)

### 思路

用到了一个性质：**二分图当且仅当图中不含奇数节点的环**

```C++
#include<bits/stdc++.h>
using namespace std;

const int N=1e5+10,M=2e5+10;

int e[M],ne[M],h[N],idx;
int n,m;
int color[N];

void add(int a,int b)
{
    e[idx]=b,ne[idx]=h[a],h[a]=idx++;
}

bool dfs(int u,int c)
{
    color[u]=c;
    
    for(int i=h[u];i!=-1;i=ne[i])
    {
        int j=e[i];
        //如果j点没有被涂
        if(!color[j])
        {
            //dfs j这个点 3-c代表1和2交替(NB！)
            if(!dfs(j,3-c))return false;
        }
        //如果两个点的颜色相同那么是错误的
        else if(color[j]==c) return false;
    }
    return true;
}


int main()
{
    memset(h,-1,sizeof h);
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        int a,b;
        cin>>a>>b;
        //无向图
        add(a,b),add(b,a);
    }
    
    bool flag=true;
    for(int i=1;i<=n;i++)
    {
        //如果这个点没有被染色，那么dfs染色
        if(!color[i])
        {
            //如果dfs返回false也就是出现错误不能构成二分图
            if(!dfs(i,1))
            {
                flag=false;
                break;
            }
        }
    }
    if(flag)puts("Yes");
    else puts("No");
    return 0;
}
```

## 匈牙利算法  
**O(nm),实际运行时间远小于O(nm),最大流算法**

给定一个二分图，求最大匹配

### 基本思路

1. 从左边开始遍历
2. 如果当前点没有匹配，那么就尝试匹配
3. 如果当前点有匹配，那么就尝试让当前点的匹配点去找其他点匹配

```C++
#include<bits/stdc++.h>
using namespace std;

const int N=510,M=1e5+10;

int h[N],e[M],ne[M],idx;
bool st[N];
int match[N];//下标表示右边的点，值表示左边的点

void add(int a,int b)
{
    e[idx]=b , ne[idx] = h[a] , h[a] = idx++;
}

int find(int x)
{
    for(int i=h[x];i!=-1;i=ne[i])
    {
        int j=e[i];
        //如果这时右边的j没有被找到
        if(!st[j])
        {
            st[j]=true;
            //如果此时match[j]没被找上或者match[j]还可以再找别人
            //这里递归后st[j]还是为true 也就是说此时match[j]无法再找到j 只能去找别人
            if(match[j]==0||find(match[j]))
            {
                //更新成x 并返回成功找到
                match[j]=x;
            
                return true;
            }
        }
    }
    return false;
}

int main()
{
    memset(h,-1,sizeof h);
    int n1,n2,m;
    cin>>n1>>n2>>m;
    while(m--)
    {
        int a,b;
        cin>>a>>b;
        add(a,b);//不选择无向图的原因是只需要看左边选择右边
        //永远都是从左边选右边
    }
    //满足的个数
    int res=0;
    //遍历所有左边的点
    for(int i=1;i<=n1;i++)
    {
        //这里的重置st是算法精髓
        //只有每次开始遍历的时候 右边所有的点都为false 这时才能完全遍历到
        memset(st,false,sizeof st);
        //如果找到合适的res++
        if(find(i))res++;
    }
    cout<<res<<endl;
    return 0;
}
```
