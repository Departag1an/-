```C++
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10;

vector<int> path; // 存储路径
int n, m;

void dfs(vector<vector<int> > &g, int x) { // 因为遍历顺序正好是1 ~ n, x相当于层数
    if (x == n) { // 前提条件是图中不存在自环, 如果存在, 需要判断是否会走已经走过的节点
        for (int i = 0; i < path.size(); i++) cout << path[i] << ' ';       
        cout << endl;
    }
    for (int i = 1; i <= n; i++) {
        if (g[x][i] == 1) {
            path.push_back(i); 
            dfs(g, i);
            path.pop_back(); // 回溯
        }
    }
}

int main () {
    cin >> n >> m;
    vector<vector<int> >graph(n + 1, vector<int> (n + 1, 0));
    for (int i = 0; i < n; i++) cout << graph[i].size() << endl;

    for (int i = 0; i < m; i++) {
        int s, t;
        cin >> s >> t;
        graph[s][t] = 1;
    }
    
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            cout << graph[i][j] << ' ';
        }
        cout << endl;
    }
    path.push_back(1);
    dfs(graph, 1);
    return 0;
}
```
